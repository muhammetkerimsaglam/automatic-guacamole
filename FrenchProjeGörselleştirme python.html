fig, axes = plt.subplots(1, 2, figsize=(14, 5))
fig.suptitle("Hedef Değişken: Hasar Sayısı (ClaimNb)", fontsize=14, fontweight="bold")
 
# Sol: ClaimNb frekansı (0,1,2,3+ gruplama)
claim_counts = df["ClaimNb"].clip(upper=3).value_counts().sort_index()
labels = ["0 hasar", "1 hasar", "2 hasar", "3+ hasar"]
colors = ["#4C72B0", "#DD8452", "#C44E52", "#8172B2"]
 
axes[0].bar(labels, claim_counts.values, color=colors, edgecolor="white", linewidth=0.8)
axes[0].set_title("Hasar Sayısı Dağılımı")
axes[0].set_ylabel("Poliçe Sayısı")
for i, v in enumerate(claim_counts.values):
    axes[0].text(i, v + 2000, f"{v:,}", ha="center", fontsize=9)
 
# Sağ: Pasta — hasar var/yok oranı
hasar_oran = df["HasClaim"].value_counts()
axes[1].pie(
    hasar_oran.values,
    labels=["Hasar Yok (%95)", "Hasar Var (%5)"],
    colors=["#4C72B0", "#DD8452"],
    autopct="%1.1f%%",
    startangle=90,
    wedgeprops={"edgecolor": "white", "linewidth": 2}
)
axes[1].set_title("Hasar Var / Yok Oranı")
 
plt.tight_layout()
plt.savefig("plot1_hedef_degisken.png", bbox_inches="tight")
plt.show()

age_hasar = (
    df.groupby("DrivAge_Bin", observed=True)["HasClaim"]
    .mean()
    .mul(100)
    .reset_index()
    .rename(columns={"HasClaim": "Hasar_Orani"})
)
 
fig, ax = plt.subplots(figsize=(9, 5))
sns.barplot(data=age_hasar, x="DrivAge_Bin", y="Hasar_Orani",hue="DrivAge_Bin",
            palette="Blues_d", edgecolor="white", ax=ax)
ax.set_title("Sürücü Yaş Grubuna Göre Hasar Oranı", fontsize=13, fontweight="bold")
ax.set_xlabel("Sürücü Yaş Grubu")
ax.set_ylabel("Hasar Oranı (%)")
ax.yaxis.set_major_formatter(mtick.FormatStrFormatter("%.1f%%"))
for bar in ax.patches:
    ax.text(
        bar.get_x() + bar.get_width() / 2,
        bar.get_height() + 0.05,
        f"{bar.get_height():.2f}%",
        ha="center", fontsize=9
    )
 
# Aktüeryal yorum notu
ax.axhline(y=df["HasClaim"].mean() * 100, color="red",
           linestyle="--", linewidth=1.2, label="Genel Ortalama")
ax.legend()
plt.tight_layout()
plt.savefig("plot2_yas_hasar.png", bbox_inches="tight")
plt.show()
print("✓ plot2 kaydedildi")

region_hasar = (
    df.groupby("Region")["HasClaim"]
    .mean()
    .mul(100)
    .sort_values(ascending=True)
    .reset_index()
    .rename(columns={"HasClaim": "Hasar_Orani"})
)
 
fig, ax = plt.subplots(figsize=(9, 8))
colors_reg = ["#e74c3c" if x > df["HasClaim"].mean() * 100 else "#4C72B0"
              for x in region_hasar["Hasar_Orani"]]
ax.barh(region_hasar["Region"], region_hasar["Hasar_Orani"],
        color=colors_reg, edgecolor="white")
ax.axvline(x=df["HasClaim"].mean() * 100, color="black",
           linestyle="--", linewidth=1.2, label="Genel Ortalama")
ax.set_title("Bölgeye Göre Hasar Oranı\n(Kırmızı = Ortalama Üstü)", fontsize=13, fontweight="bold")
ax.set_xlabel("Hasar Oranı (%)")
ax.xaxis.set_major_formatter(mtick.FormatStrFormatter("%.1f%%"))
ax.legend()
plt.tight_layout()
plt.savefig("plot6_bolge.png", bbox_inches="tight")
